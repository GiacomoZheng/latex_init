%% for (), [] and \{\}
\def\oparen{\left\lparen} \def\cparen{\right\rparen} % ()
\def\obrack{\left\lbrack} \def\cbrack{\right\rbrack} % []
\def\{{\left\lbrace} \def\}{\right\rbrace} % \{\}
{
	\catcode`\(=13 \gdef({\oparen} \catcode`\)=13 \gdef){\cparen}
	\catcode`\[=13 \gdef[{\obrack} \catcode`\]=13 \gdef]{\cbrack}
}
\everydisplay={
	\catcode`\(=13 \catcode`\)=13 \catcode`\[=13 \catcode`\]=13
}

%% for amsmath tilte
\let\origmaketitle\maketitle
\def\maketitle{
	\begingroup
	\def\uppercasenonmath##1{} % this disables uppercasing title
	\let\MakeUppercase\relax % this disables uppercasing authors
	\origmaketitle
	\endgroup
}
